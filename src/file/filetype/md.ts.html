// @ts-ignore
const renderArticle = (
  mdArticle: any,
  config: PageSettings & { title: string; siteName: string }
) => {
  const pageName = mdArticle.name;

  return [
    "html",
    ["header", config],
    [
      "body",
      ["Sidebar"],
      [
        "div",
        { class: "site-body" },
        [
          "main",
          [
            "article",
            { class: "wikipage" },
            fileHasTitle ?? ["h1", { class: "title-top", pageName }],
            mdArticle,
          ],
        ],
        [
          "div",
          { class: "article-rhs-container" },
          [
            "div",
            { class: "article-rhs" },
            ["PageMap"],
            ["GitHistoryTable"],
            ["PrevNextUpButtons"],
          ],
        ],
      ],
    ],
  ];
};
